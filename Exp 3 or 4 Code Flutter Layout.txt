import 'package:flutter/material.dart';

void main() {
  runApp(const MyLayoutApp());
}

class MyLayoutApp extends StatelessWidget {
  const MyLayoutApp({super.key});

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      debugShowCheckedModeBanner: false,
      home: Scaffold(
        appBar: AppBar(
          title: const Text("Flutter Layout Example"),
        ),
        body: Column(
          children: [

            // Top Image (Pexels)
            Image.network(
              'https://images.pexels.com/photos/36286811/pexels-photo-36286811.jpeg',
              height: 400,
              width: double.infinity,
              fit: BoxFit.contain,
            ),

            const SizedBox(height: 20),

            // Row with two containers
            Row(
              mainAxisAlignment: MainAxisAlignment.spaceEvenly,
              children: [
                Container(
                  padding: const EdgeInsets.all(20),
                  color: Colors.blueAccent,
                  child: const Text(
                    "Left Box",
                    style: TextStyle(
                      color: Colors.white,
                      fontSize: 18,
                    ),
                  ),
                ),
                Container(
                  padding: const EdgeInsets.all(20),
                  color: Colors.green,
                  child: const Text(
                    "Right Box",
                    style: TextStyle(
                      color: Colors.white,
                      fontSize: 18,
                    ),
                  ),
                ),
              ],
            ),

            const SizedBox(height: 20),

            // Bottom Text
            const Padding(
              padding: EdgeInsets.all(16.0),
              child: Text(
                "This is a simple layout using Column, Row, and Image widgets.",
                style: TextStyle(fontSize: 16),
                textAlign: TextAlign.center,
              ),
            ),

          ],
        ),
      ),
    );
  }
}
